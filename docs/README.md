```python
from asyncio import run
from emojidb import EmojiDBClient


async def main():
    query = "love"
    async with EmojiDBClient() as client:
        for emoji, info in await client.search_for_emojis(query):
            print(emoji, info)


if __name__ == "__main__":
    run(main)
```
